create table public.categories (

  slug text not null,
  created_at timestamp with time zone null default now(),
  constraint categories_pkey primary key (id),
  constraint categories_slug_key unique (slug)

create table public.collection_products (
  collection_id bigint not null,
  product_id bigint not null,
  added_at timestamp with time zone null default now(),
  constraint collection_products_pkey primary key (collection_id, product_id),
  constraint collection_products_collection_id_fkey foreign KEY (collection_id) references collections (id) on delete CASCADE,
  constraint collection_products_product_id_fkey foreign KEY (product_id) references products (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.collections (
  id bigint generated by default as identity not null,
  title text not null,
  subtitle text null,
  image_url text null,
  bg_color text null,
  is_active boolean null default true,
  created_at timestamp with time zone null default now(),
  type text null default 'watch'::text,
  constraint collections_pkey primary key (id),
  constraint collections_type_check check (

      type = any (array['watch'::text, 'perfume'::text])
    )

create table public.inventory_logs (
  id bigint generated by default as identity not null,

  change_amount integer not null,
  reason text null,
  created_at timestamp with time zone null default now(),
  constraint inventory_logs_pkey primary key (id),
  constraint inventory_logs_variant_id_fkey foreign KEY (variant_id) references product_variants (id)
) TABLESPACE pg_default;

create table public.order_items (
  id bigint generated by default as identity not null,
  order_id uuid null,
  variant_id bigint null,
  quantity integer not null default 1,
  unit_price numeric(10, 2) not null,
  constraint order_items_pkey primary key (id),
  constraint order_items_order_id_fkey foreign KEY (order_id) references orders (id) on delete CASCADE,
  constraint order_items_variant_id_fkey foreign KEY (variant_id) references product_variants (id)
) TABLESPACE pg_default;

create table public.orders (
  id uuid not null default gen_random_uuid (),
  user_id uuid null,
  customer_name text not null,
  customer_email text null,
  customer_phone text not null,
  shipping_address jsonb not null,
  status text null default 'pending'::text,
  total_amount numeric(10, 2) not null,
  voucher_code text null,
  discount_amount numeric(10, 2) null default 0,
  created_at timestamp with time zone null default now(),
  city text not null default 'Unknown'::text,
  postal_code text null,
  constraint orders_pkey primary key (id),
  constraint orders_user_id_fkey foreign KEY (user_id) references auth.users (id),
  constraint orders_status_check check (
    (
      status = any (
        array[
          'pending'::text,
          'confirmed'::text,
          'on route'::text,
          'delivered'::text,
          'cancelled'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

create table public.product_images (
  id bigint generated by default as identity not null,
  product_id bigint null,
  image_url text not null,
  is_primary boolean null default false,
  display_order integer null default 0,
  variant_id bigint null,
  constraint product_images_pkey primary key (id),
  constraint product_images_product_id_fkey foreign KEY (product_id) references products (id) on delete CASCADE,
  constraint product_images_variant_id_fkey foreign KEY (variant_id) references product_variants (id) on delete CASCADE,
  constraint check_variant_belongs_to_product check (true)
) TABLESPACE pg_default;

create index IF not exists idx_product_images_variant on public.product_images using btree (variant_id) TABLESPACE pg_default;

create table public.product_variants (
  id bigint generated by default as identity not null,
  product_id bigint null,
  size text null,
  color text null,
  sku text not null,
  stock_quantity integer null default 0,
  price numeric(10, 2) not null,
  sale_price numeric(10, 2) null,
  is_on_sale boolean null default false,
  created_at timestamp with time zone null default now(),
  is_primary boolean null,
  constraint product_variants_pkey primary key (id),
  constraint product_variants_sku_key unique (sku),
  constraint product_variants_product_id_fkey foreign KEY (product_id) references products (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.products (
  id bigint generated by default as identity not null,
  category_id bigint null,
  title text not null,
  description text null,
  base_price numeric(10, 2) not null,
  is_active boolean null default true,
  created_at timestamp with time zone null default now(),
  brand text null,
  constraint products_pkey primary key (id),
  constraint products_category_id_fkey foreign KEY (category_id) references categories (id)
) TABLESPACE pg_default;

create table public.profiles (
  id uuid not null,
  full_name text null,
  phone_number text null,
  avatar_url text null,
  created_at timestamp with time zone null default timezone ('utc'::text, now()),
  constraint profiles_pkey primary key (id),
  constraint profiles_id_fkey foreign KEY (id) references auth.users (id)
) TABLESPACE pg_default;

create table public.vouchers (
  id bigint generated by default as identity not null,
  code text not null,
  discount_type text null,
  discount_value numeric(10, 2) not null,
  min_order_value numeric(10, 2) null default 0,
  usage_limit integer null,
  used_count integer null default 0,
  expires_at timestamp with time zone null,
  is_active boolean null default true,
  constraint vouchers_pkey primary key (id),
  constraint vouchers_code_key unique (code),
  constraint vouchers_discount_type_check check (
    (
      discount_type = any (array['percentage'::text, 'fixed_amount'::text])
    )
  )
) TABLESPACE pg_default;
